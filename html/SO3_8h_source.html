22 /**
23  Class SO3 is actually represented by the quaternion,
24  which use 4 paraments to present a 3 dimesion rotation matrix,
25  since 3D rotation matrices are members of the Special Orthogonal Lie group SO3.
26 
27  Every rotation in the 3D euclidean space can be represented by a rotation with one direction.
28  Consider a rotation with direction (a,b,c)^T and angle theta in radians.
29  w -- cos(theta/2)
30  x -- a*sin(theta/2)
31  y -- b*sin(theta/2)
32  z -- c*sin(theta/2)
33  this ensures that x^2+y^2+z^2+w^2=1.
34  A quaternion q(x,y,z,w) is used to present a 3d rotation.
35 
36  it can be constructed by a matrix:(tested TooN,Eigen)
37  @code
38  Eigen::Matrix3d M;
39  SO3<double> so3;
40  // You should ensure that both the SO3 and matrix has the same precision(double or float)
41  so3.fromMatrix(M);
42  @endcode
43 
44  This group can be parameterised by three numbers (a vector in the space of Lie Algebra).
45  Here uses r=(rx,ry,rz)^T to present a rotation,
46  where its direction is the axis of rotation and its length is the angle of rotation in radians.
47  It is also used by TooN (also know as PTAM) and Sophus.
48  @code
49  SO3d so3=SO3d::exp(Point3d(1,1,1)); //construct from 3 parament
50  cout<<"Three paraments to present a rotation:"<<so3.ln()<<endl;
51  @endcode
52 
53  */

84  Point3_<Precision> log()const
85  {
86  const Precision squared_w = w*w;
87  const Precision n = sqrt(x*x+y*y+z*z);
88 
89  Precision A_inv;
90  // Atan-based log thanks to
91  //
92  // C. Hertzberg et al.:
93  // "Integrating Generic Sensor Fusion Algorithms with Sound State
94  // Representation through Encapsulation of Manifolds"
95  // Information Fusion, 2011
96 
97  if (n < NEAR_ZERO)
98  {
99  //If n is too small
100  A_inv = 2./w - 2.*(1.0-squared_w)/(w*squared_w);
101  }
102  else
103  {
104  if (fabs(w)<NEAR_ZERO)
105  {
106  //If w is too small
107  if (w>0)
108  {
109  A_inv = M_PI/n;
110  }
111  else
112  {
113  A_inv = -M_PI/n;
114  }
115  }
116  else
117  A_inv = 2*atan(n/w)/n;
118  }
119  return Point3_<Precision>(x*A_inv,y*A_inv,z*A_inv);

218  /// M is curved as follow
219  /// |m0 m1 m2|
220  /// |m3 m4 m5|
221  /// |m6 m7 m8|
222  inline bool fromMatrix(const Precision* m)
223  {
224  if(false)
225  {
226  Precision tr=m[0]+m[4]+m[8];
227  w=0.5*sqrt(tr+1);
228  Precision oneover4w=0.25/w;
229  x=(m[7]-m[5])*oneover4w;
230  y=(m[2]-m[6])*oneover4w;
231  z=(m[3]-m[1])*oneover4w;
232  if(w<0){x*=-1;y*=-1;z*=-1;w*=-1;}
233  return true;
234  }
235  else
236  {
237  Precision& q0=w;
238  Precision& q1=x;
239  Precision& q2=y;
240  Precision& q3=z;
241  const Precision &r11=m[0],&r12=m[1],&r13=m[2],
242  &r21=m[3],&r22=m[4],&r23=m[5],
243  &r31=m[6],&r32=m[7],&r33=m[8];
244  q0 = ( r11 + r22 + r33 + 1.0f) / 4.0f;
245  q1 = ( r11 - r22 - r33 + 1.0f) / 4.0f;
246  q2 = (-r11 + r22 - r33 + 1.0f) / 4.0f;
247  q3 = (-r11 - r22 + r33 + 1.0f) / 4.0f;
248  if(q0 < 0.0f) q0 = 0.0f;
249  if(q1 < 0.0f) q1 = 0.0f;
250  if(q2 < 0.0f) q2 = 0.0f;
251  if(q3 < 0.0f) q3 = 0.0f;
252  q0 = sqrt(q0);
253  q1 = sqrt(q1);
254  q2 = sqrt(q2);
255  q3 = sqrt(q3);
256  if(q0 >= q1 && q0 >= q2 && q0 >= q3) {
257  q0 *= +1.0f;
258  q1 *= SIGN(r32 - r23);
259  q2 *= SIGN(r13 - r31);
260  q3 *= SIGN(r21 - r12);
261  } else if(q1 >= q0 && q1 >= q2 && q1 >= q3) {
262  q0 *= SIGN(r32 - r23);
263  q1 *= +1.0f;
264  q2 *= SIGN(r21 + r12);
265  q3 *= SIGN(r13 + r31);
266  } else if(q2 >= q0 && q2 >= q1 && q2 >= q3) {
267  q0 *= SIGN(r13 - r31);
268  q1 *= SIGN(r21 + r12);
269  q2 *= +1.0f;
270  q3 *= SIGN(r32 + r23);
271  } else if(q3 >= q0 && q3 >= q1 && q3 >= q2) {
272  q0 *= SIGN(r21 - r12);
273  q1 *= SIGN(r31 + r13);
274  q2 *= SIGN(r32 + r23);
275  q3 *= +1.0f;
276  } else {
277  printf("coding error\n");
278  return false;
279  }
280  Precision r = sqrt(q0*q0+q1*q1+q2*q2+q3*q3);
281  q0 /= r;
282  q1 /= r;
283  q2 /= r;
284  q3 /= r;
285  }
286  return true;
287  }

391  void FromEuler(const Precision& pitch,const Precision& yaw,const Precision& roll)
392  {
393  // Basically we create 3 Quaternions, one for pitch, one for yaw, one for roll
394  // and multiply those together.
395  // the calculation below does the same, just shorter
396  Precision piover360=0.5;//3.1415926/360.0;
397  Precision p = pitch * piover360;
398  Precision y = yaw * piover360;
399  Precision r = roll * piover360;
400 
401 
402  Precision sinp = sin(p);
403  Precision siny = sin(y);
404  Precision sinr = sin(r);
405  Precision cosp = cos(p);
406  Precision cosy = cos(y);
407  Precision cosr = cos(r);
408 
409

411  this->y = cosr * sinp * cosy + sinr * cosp * siny;
412  this->z = cosr * cosp * siny - sinr * sinp * cosy;
