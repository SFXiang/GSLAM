168 inline double Timer::do_leave(const char *func_name)
169 {
170  using namespace std;
171  const double tim = Tac();
172 
173  const string s = func_name;
174  TCallData *d=NULL;
175  {
176  std::unique_lock<std::mutex> lock(mMutex);
177  d = &m_data[s];
178  }
179  if (!d->open_calls.empty())
180  {
181  const double At = tim - d->open_calls.top();
182  d->open_calls.pop();
183 
184  d->mean_t+=At;
185  if (d->n_calls==1)
186  {
187  d->min_t= At;
188  d->max_t= At;
189  }
190  else
191  {
192  if (d->min_t>At) d->min_t = At;
193  if (d->max_t<At) d->max_t = At;
194  }
195  return At;
196  }
197  else return 0; // This shouldn't happen!
198 }
