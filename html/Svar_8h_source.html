585 inline const char* Svar::MatchingEndBrace(const char* str, char flag) {
586  char endflag = '}';

588  endflag = ')';

605 inline std::vector<std::string> Svar::ChopAndUnquoteString(std::string s) {
606  using namespace std;
607  vector<string> v;
608  string::size_type nPos = 0;
609  string::size_type nLength = s.length();
610  while (1) {
611  string sTarget;
612  char cDelim;
613  // Get rid of leading whitespace:
614  while ((nPos < nLength) && (s[nPos] == ' ')) nPos++;
615  if (nPos == nLength) return v;
616 
617  // First non-whitespace char...
618  if (s[nPos] != '\"')
619  cDelim = ' ';
620  else {
621  cDelim = '\"';
622  nPos++;
623  }
624  for (; nPos < nLength; ++nPos) {
625  char c = s[nPos];
626  if (c == cDelim) break;
627  if (cDelim == '"' && nPos + 1 < nLength && c == '\\') {
628  char escaped = s[++nPos];
629  switch (escaped) {
630  case 'n':
631  c = '\n';
632  break;
633  case 'r':
634  c = '\r';
635  break;
636  case 't':
637  c = '\t';
638  break;
639  default:
640  c = escaped;
641  break;
642  }
643  }
644  sTarget += c;
645  }
646  v.push_back(sTarget);
647 
648  if (cDelim == '\"') nPos++;
649  }
650 }

763 inline std::string Svar::expandVal(std::string val, char flag) {
764  using namespace std;
765  string expanded = val;
766 
767  while (true) {
768  const char* brace = FirstOpenBrace(expanded.c_str(), flag);
769  if (brace) {
770  const char* endbrace = MatchingEndBrace(brace, flag);
771  if (endbrace) {
772  ostringstream oss;
773  oss << std::string(expanded.c_str(), brace - 1);
774 
775  const string inexpand =
776  expandVal(std::string(brace + 1, endbrace), flag);
777  if (exist(inexpand)) {
778  oss << getvar(inexpand);
779  } else {
780  printf(
781  "Unabled to expand: [%s].\nMake sure it is defined and "
782  "terminated with a semi-colon.\n",
783  inexpand.c_str());
784  oss << "#";
785  }
786 
787  oss << std::string(endbrace + 1, expanded.c_str() + expanded.length());
788  expanded = oss.str();
789  continue;
790  }
791  }
792  break;
793  }

802 inline bool Svar::setvar(std::string s) {
803  using namespace std;
804  // Execution failed. Maybe its an assignment.
805  string::size_type n;
806  n = s.find("=");
807  bool shouldOverwrite = true;
808 
809  if (n != string::npos) {
810  string var = s.substr(0, n);
811  string val = s.substr(n + 1);
812 
813  // Strip whitespace from around var;
814  string::size_type s = 0, e = var.length() - 1;
815  if ('?' == var[e]) {
816  e--;
817  shouldOverwrite = false;
818  }
819  for (; isspace(var[s]) && s < var.length(); s++) {
820  }
821  if (s == var.length()) // All whitespace before the `='?
822  return false;
823  for (; isspace(var[e]); e--) {
824  }
825  if (e >= s) {
826  var = var.substr(s, e - s + 1);
827 
828  // Strip whitespace from around val;
829  s = 0, e = val.length() - 1;
830  for (; isspace(val[s]) && s < val.length(); s++) {
831  }
832  if (s < val.length()) {
833  for (; isspace(val[e]); e--) {
834  }
835  val = val.substr(s, e - s + 1);
836  } else
837  val = "";
838 
839  insert(var, val, shouldOverwrite);
840  return true;
841  }
842  }
843 
844  return false;
845 }

904 inline bool Svar::ParseLine(std::string s, bool bSilentFailure) {
905  using namespace std;
906  if (s == "") return 0;
907  int& collectFlag = GetInt("Svar.Collecting", 0);
908  if (collectFlag) {
909  istringstream ist(s);
910  string sCommand;
911  ist >> sCommand;
912  if (sCommand == "endif" || sCommand == "fi") command().Call("endif");
913  if (sCommand == "else")
914  command().Call("else");
915  else if (sCommand == "endfunction")
916  command().Call("endfunction");
917  else if (sCommand == ".") {
918  command().Call(".", ist.str());
919  } else
920  command().Call(".", s);
921  return 0;
922  }
923  s = UncommentString(expandVal(s, '{'));
924  s = UncommentString(expandVal(s, '('));
925  if (s == "") return 0;
926 
927  // Old ParseLine code follows, here no braces can be left (unless in arg.)
928  istringstream ist(s);
929 
930  string sCommand;
931  string sParams;
932 
933  // Get the first token (the command name)
934  ist >> sCommand;
935  if (sCommand == "") return 0;
936 
937  // Get everything else (the arguments)...
938 
939  // Remove any whitespace
940  ist >> ws;
941  getline(ist, sParams);
942 
943  // Attempt to execute command
944  if (command().Call(sCommand, sParams)) return true;
945 
946  if (setvar(s)) return 1;
947 
948  if (!bSilentFailure)
949  cerr << "? GUI_impl::ParseLine: Unknown command \"" << sCommand
950  << "\" or invalid assignment." << endl;
951  return 0;
952 }

961  insert("Svar.ParsingFile", parsingFile, true);

964  while (getline(is, buffer) && shouldParse) {
965  // Lines ending with '\' are taken as continuing on the next line.
