85 inline bool UndistorterImpl::prepareReMap()
86 {
87  using namespace std;
88  // check camera model
89  if( !(camera_in.isValid() && camera_out.isValid()) )
90  {
91  valid = false;
92 // cout<<("Undistorter does not get vallid camera.");
93  return false;
94  }
95  // Prepare remap
96  cout << "Undistorter:\n";
97  cout << " Camera IN : " << camera_in.info() << endl;
98  cout << " Camera OUT: " << camera_out.info() << endl << endl;
99  int size=camera_out.width() * camera_out.height();
100  remapX = new float[size];
101  remapY = new float[size];
102  remapFast = new int[size];
103 
104  remapIdx = new int [size*4];
105  remapCoef = new float[size*4];
106  Point3d world_pose;Point2d im_pose;
107  int w_in=camera_in.width(),h_in=camera_in.height();
108  for(int y=0,yend=camera_out.height(); y<yend; y++)
109  for(int x=0,xend=camera_out.width(); x<xend; x++)
110  {
111  int i = y*xend+x;
112 
113  world_pose=camera_out.UnProject(Point2d(x,y));
114  im_pose = camera_in.Project(world_pose);
115 
116  if(im_pose.x<0 || im_pose.y<0 ||
117  im_pose.x>=w_in||im_pose.y>=h_in)
118  {
119  remapX[i] = -1;
120  remapY[i] = -1;
121  remapFast[i] = -1;
122 
123  remapIdx[i*4+0] = 0;
124  remapIdx[i*4+1] = 0;
125  remapIdx[i*4+2] = 0;
126  remapIdx[i*4+3] = 0;
127 
128  remapCoef[i*4+0] = 0;
129  remapCoef[i*4+1] = 0;
130  remapCoef[i*4+2] = 0;
131  remapCoef[i*4+3] = 0;
132  continue;
133  }
134 
135  {
136  remapX[i] = im_pose.x;
137  remapY[i] = im_pose.y;
138  remapFast[i] = (int)im_pose.x+w_in*(int)im_pose.y;
139  }
140 
141  // calculate fast bi-linear interpolation indices & coefficients
142  {
143  float xx = remapX[i];
144  float yy = remapY[i];
145 
146  if( xx < 0.0 ) continue;
147 
148  // get integer and rational parts
149  int xxi = xx;
150  int yyi = yy;
151  xx -= xxi;
152  yy -= yyi;
153  float xxyy = xx*yy;
154 
155  remapIdx[i*4+0] = yyi*w_in + xxi;
156  remapIdx[i*4+1] = yyi*w_in + xxi + 1;
157  remapIdx[i*4+2] = (yyi+1)*w_in + xxi;
158  remapIdx[i*4+3] = (yyi+1)*w_in + xxi + 1;
159 
160  remapCoef[i*4+0] = 1-xx-yy+xxyy;
161  remapCoef[i*4+1] = xx-xxyy;
162  remapCoef[i*4+2] = yy-xxyy;
163  remapCoef[i*4+3] = xxyy;
164  }
165  }
166  valid=true;
167  return true;
168 }
