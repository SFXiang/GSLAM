8 /**
9  Represent a three-dimensional Euclidean transformation (a rotation and a translation).
10  | r00 r01 r02 t0 |
11 M = | R T |= | r10 r11 r12 t1 | = | Rx Ry Rz T |
12  | 0 1 | | r20 r21 r22 t2 | | 0 0 0 1 |
13  | 0 0 0 1 |
14 Here R=(Rx,Ry,Rz) is the rotation , T=(x,y,z)^T is the translation.
15 
16 For the matrix M (only if the matrix means a camera to world transform):
17 Rx means the direction of X axis;
18 Ry means the direction of Y axis;
19 Rz means the direction of Z axis;
20 T means the translation of this coordinate.
21 
22 
23 The coordinates of point P=(X,Y,Z)^T in this coordinate can be computed by left-multiply the matrix M:
24 @code
25 SE3<double> C2W; // Camera to world SE3
26 Point3d P_cam(1,2,3); // Coordinates in the camera coordinate
27 cout<<"Coordinates in the world coordinate:"<<C2W*P_cam<<endl;
28 
29 SE3<double> W2C=C2W.inv(); // It also can present a world to camera transform
30 Point3d P_w(4,5,6); // A point in the world
31 cout<<"The coodinates in the camera is "<<W2C*P_w;
32 @endcode
33 
34 It should be noticed that when a SE3 is representing the world-to-camera transform,
35 the function <code>get_translation()</code> does not acqually return the translation.
36 
37  */

189  inline Array_<Precision,6> log()const
190  {
191  Array_<Precision,6> result;
192  const auto& l=my_rotation;
193  const auto& t=my_translation;
194  const Precision squared_w = l.w*l.w;
195  const Precision n = sqrt(l.x*l.x+l.y*l.y+l.z*l.z);
196 
197  Precision A_inv;
198  // Atan-based log thanks to
199  //
200  // C. Hertzberg et al.:
201  // "Integrating Generic Sensor Fusion Algorithms with Sound State
202  // Representation through Encapsulation of Manifolds"
203  // Information Fusion, 2011
204 
205  if (n < NEAR_ZERO)
206  {
207  //If n is too small
208  A_inv = 2./l.w - 2.*(1.0-squared_w)/(l.w*squared_w);
209  Point3_<Precision> r(l.x*A_inv,l.y*A_inv,l.z*A_inv);
210  Point3_<Precision> p=t-0.5*r.cross(t)+static_cast<Precision>(1. / 12.)*r.cross(r.cross(t));
211  *(Point3_<Precision>*)&result.data=p;
212  *(Point3_<Precision>*)&result.data[3]=r;
213  }
214  else
215  {
216  if (fabs(l.w)<NEAR_ZERO)
217  {
218  //If w is too small
219  if (l.w>0)
220  {
221  A_inv = M_PI/n;
222  }
223  else
224  {
225  A_inv = -M_PI/n;
226  }
227  }
228  else
229  A_inv = 2*atan(n/l.w)/n;
230 
231  auto theta=A_inv*n;
232  Point3_<Precision> r(l.x*A_inv,l.y*A_inv,l.z*A_inv);
233  Point3_<Precision> a=r/theta;
234  Point3_<Precision> p=t-0.5*r.cross(t)+(1-theta/(2*tan(0.5*theta)))*a.cross(a.cross(t));
235  *(Point3_<Precision>*)&result.data=p;
236  *(Point3_<Precision>*)&result.data[3]=r;
237  }
238  return result;
239  }

241  template <typename Scalar>
242  static inline SE3<Scalar> exp(const Array_<Scalar,6>& l)
243  {
244  Point3_<Scalar> p(l.data[0],l.data[1],l.data[2]);
245  Point3_<Scalar> r(l.data[3],l.data[4],l.data[5]);
246  Scalar theta_sq = r.dot(r);
247  Scalar theta = sqrt(theta_sq);
248  Scalar half_theta = static_cast<Scalar>(0.5) * theta;
249 
250  Scalar imag_factor;
251  Scalar real_factor;
252 
253  if (theta < static_cast<Scalar>(1e-10)) {
254  Scalar theta_po4 = theta_sq * theta_sq;
255  imag_factor = static_cast<Scalar>(0.5) -
256  static_cast<Scalar>(1.0 / 48.0) * theta_sq +
257  static_cast<Scalar>(1.0 / 3840.0) * theta_po4;
258  real_factor = static_cast<Scalar>(1) -
259  static_cast<Scalar>(0.5) * theta_sq +
260  static_cast<Scalar>(1.0 / 384.0) * theta_po4;
261  } else {
262  Scalar sin_half_theta = sin(half_theta);
263  imag_factor = sin_half_theta / theta;
264  real_factor = cos(half_theta);
265  }
266 
267  SO3<Scalar> R( imag_factor * r.x, imag_factor * r.y,imag_factor * r.z,real_factor);
268  auto t= p+(1-cos(theta))/theta_sq*r.cross(p)+
269  (theta-sin(theta))/(theta_sq*theta)*r.cross(r.cross(p));
270  return SE3<Scalar>(R,t);
271  }
