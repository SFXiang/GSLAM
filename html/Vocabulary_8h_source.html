567 inline double L1Scoring::score(const BowVector &v1, const BowVector &v2) const
568 {
569  BowVector::const_iterator v1_it, v2_it;
570  const BowVector::const_iterator v1_end = v1.end();
571  const BowVector::const_iterator v2_end = v2.end();
572 
573  v1_it = v1.begin();
574  v2_it = v2.begin();
575 
576  double score = 0;
577 
578  while(v1_it != v1_end && v2_it != v2_end)
579  {
580  const WordValue& vi = v1_it->second;
581  const WordValue& wi = v2_it->second;
582 
583  if(v1_it->first == v2_it->first)
584  {
585  score += fabs(vi - wi) - fabs(vi) - fabs(wi);
586 
587  // move v1 and v2 forward
588  ++v1_it;
589  ++v2_it;
590  }
591  else if(v1_it->first < v2_it->first)
592  {
593  // move v1 forward
594  v1_it = v1.lower_bound(v2_it->first);
595  // v1_it = (first element >= v2_it.id)
596  }
597  else
598  {
599  // move v2 forward
600  v2_it = v2.lower_bound(v1_it->first);
601  // v2_it = (first element >= v1_it.id)
602  }
603  }
604 
605  // ||v - w||_{L1} = 2 + Sum(|v_i - w_i| - |v_i| - |w_i|)
606  // for all i | v_i != 0 and w_i != 0
607  // (Nister, 2006)
608  // scaled_||v - w||_{L1} = 1 - 0.5 * ||v - w||_{L1}
609  score = -score/2.0;
610 
611  return score; // [0..1]
612 }

617 inline double L2Scoring::score(const BowVector &v1, const BowVector &v2) const
618 {
619  BowVector::const_iterator v1_it, v2_it;
620  const BowVector::const_iterator v1_end = v1.end();
621  const BowVector::const_iterator v2_end = v2.end();
622 
623  v1_it = v1.begin();
624  v2_it = v2.begin();
625 
626  double score = 0;
627 
628  while(v1_it != v1_end && v2_it != v2_end)
629  {
630  const WordValue& vi = v1_it->second;
631  const WordValue& wi = v2_it->second;
632 
633  if(v1_it->first == v2_it->first)
634  {
635  score += vi * wi;
636 
637  // move v1 and v2 forward
638  ++v1_it;
639  ++v2_it;
640  }
641  else if(v1_it->first < v2_it->first)
642  {
643  // move v1 forward
644  v1_it = v1.lower_bound(v2_it->first);
645  // v1_it = (first element >= v2_it.id)
646  }
647  else
648  {
649  // move v2 forward
650  v2_it = v2.lower_bound(v1_it->first);
651  // v2_it = (first element >= v1_it.id)
652  }
653  }
654 
655  // ||v - w||_{L2} = sqrt( 2 - 2 * Sum(v_i * w_i) )
656  // for all i | v_i != 0 and w_i != 0 )
657  // (Nister, 2006)
658  if(score >= 1) // rounding errors
659  score = 1.0;
660  else
661  score = 1.0 - sqrt(1.0 - score); // [0..1]
662 
663  return score;
664 }

669 inline double ChiSquareScoring::score(const BowVector &v1, const BowVector &v2)
670  const
671 {
672  BowVector::const_iterator v1_it, v2_it;
673  const BowVector::const_iterator v1_end = v1.end();
674  const BowVector::const_iterator v2_end = v2.end();
675 
676  v1_it = v1.begin();
677  v2_it = v2.begin();
678 
679  double score = 0;
680 
681  // all the items are taken into account
682 
683  while(v1_it != v1_end && v2_it != v2_end)
684  {
685  const WordValue& vi = v1_it->second;
686  const WordValue& wi = v2_it->second;
687 
688  if(v1_it->first == v2_it->first)
689  {
690  // (v-w)^2/(v+w) - v - w = -4 vw/(v+w)
691  // we move the -4 out
692  if(vi + wi != 0.0) score += vi * wi / (vi + wi);
693 
694  // move v1 and v2 forward
695  ++v1_it;
696  ++v2_it;
697  }
698  else if(v1_it->first < v2_it->first)
699  {
700  // move v1 forward
701  v1_it = v1.lower_bound(v2_it->first);
702  }
703  else
704  {
705  // move v2 forward
706  v2_it = v2.lower_bound(v1_it->first);
707  }
708  }
709 
710  // this takes the -4 into account
711  score = 2. * score; // [0..1]
712 
713  return score;
714 }
