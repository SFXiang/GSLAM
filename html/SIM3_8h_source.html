8 /**
9 Represent a three-dimensional similarity transformation,7 degrees of freedom
10 a rotation: R (Here is represented by 4 paraments quaternion:Class SO3)--3 DoF
11 a translation: T (a 3 paraments vector)--3 DoF
12 a scale: s --1 DoF
13 It is the equivalent of a 4*4 Matrix:
14  | R*s T^T|
15  | 0 1 |
16 
17  This transformation is a member of the Lie group SIM3.
18  These can be parameterised with seven numbers (in the space of the Lie Algebra).
19  In this class, the first three parameters are a translation vector,
20  while the second three are a rotation vector,whose direction is the axis of rotation,
21  and length the amount of rotation (in radians), as for SO3.
22  The seventh parameter is the log of the scale of the transformation.
23 
24 **/

134 /// internal function that calculates the coefficients for the Rodrigues formula for SIM3 translation
135 template <typename Precision>
136 inline Point3_<Precision> compute_rodrigues_coefficients_sim3( const Precision & s, const Precision & t ){
137 
138  Point3_<Precision> coeff;
139  const Precision es = exp(s);
140 
141  // 4 cases for s -> 0 and/or theta -> 0
142  // the Taylor expansions were calculated with Maple 12 and truncated at the 3rd power,
143  // such that eps^3 < 1e-18 which results in approximately 1 + eps^3 = 1
144  static const Precision eps = 1e-6;
145 
146  if(fabs(s) < eps && fabs(t) < eps){
147  coeff[0] = 1 + s/2 + s*s/6;
148  coeff[1] = 1/2 + s/3 - t*t/24 + s*s/8;
149  coeff[2] = 1/6 + s/8 - t*t/120 + s*s/20;
150  } else if(fabs(s) < eps) {
151  coeff[0] = 1 + s/2 + s*s/6;
152  coeff[1] = (1-cos(t))/(t*t) + (sin(t)-cos(t)*t)*s/(t*t*t)+(2*sin(t)*t-t*t*cos(t)-2+2*cos(t))*s*s/(2*t*t*t*t);
153  coeff[2] = (t-sin(t))/(t*t*t) - (-t*t-2+2*cos(t)+2*sin(t)*t)*s/(2*t*t*t*t) - (-t*t*t+6*cos(t)*t+3*sin(t)*t*t-6*sin(t))*s*s/(6*t*t*t*t*t);
154  } else if(fabs(t) < eps) {
155  coeff[0] = (es - 1)/s;
156  coeff[1] = (s*es+1-es)/(s*s) - (6*s*es+6-6*es+es*s*s*s-3*es*s*s)*t*t/(6*s*s*s*s);
157  coeff[2] = (es*s*s-2*s*es+2*es-2)/(2*s*s*s) - (es*s*s*s*s-4*es*s*s*s+12*es*s*s-24*s*es+24*es-24)*t*t/(24*s*s*s*s*s);
158  } else {
159  const Precision a = es * sin(t);
160  const Precision b = es * cos(t);
161  const Precision inv_s_theta = 1/(s*s + t*t);
162 
163  coeff[0] = (es - 1)/s;
164  coeff[1] = (a*s + (1-b)*t) * inv_s_theta / t;
165  coeff[2] = (coeff[0] - ((b-1)*s + a*t) * inv_s_theta) / (t*t);
166  }
167 
168  return coeff;
169 }
