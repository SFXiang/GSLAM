149  static SPtr<Estimator> create(std::string pluginName = "") {
150  funcCreateEstimatorInstance createFunc =
151  SvarWithType<funcCreateEstimatorInstance>::instance()["Default"];
152  if (createFunc) return createFunc();
153 
154  if (pluginName.empty()) {
155  pluginName = svar.GetString("EstimatorPlugin", "libgslam_estimator");
156  }
157  SPtr<SharedLibrary> plugin = Registry::get(pluginName);
158  if (!plugin) return SPtr<Estimator>();
159  createFunc = (funcCreateEstimatorInstance)plugin->getSymbol(
160  "createEstimatorInstance");
161  if (!createFunc)
