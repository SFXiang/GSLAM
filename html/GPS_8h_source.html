47  ** Convert WGS84 lon,lat,alt data to ECEF data (Earth Centered Earth Fixed)
48  ** @param lat Latitude in degree
49  ** @param lon Longitude in degree
50  ** @param alt Altitude relative to the WGS84 ellipsoid
51  ** @return ECEF corresponding coordinates
52  **
53  ** http://fr.mathworks.com/matlabcentral/newsreader/view_thread/142629
54  **/
55  static pi::Point3d GPS2XYZ(double lat,double lon,double alt)
56  {
57  const double clat = cos( D2R(lat) );
58  const double slat = sin( D2R(lat) );
59  const double clon = cos( D2R(lon) );
60  const double slon = sin( D2R(lon) );
61 
62  const double a2 = Square(WGS84_A);
63  const double b2 = Square(WGS84_B);
64 
65  const double L = 1.0 / sqrt(a2 * Square(clat) + b2 * Square(slat));
66  const double x = (a2 * L + alt) * clat * clon;
67  const double y = (a2 * L + alt) * clat * slon;
68  const double z = (b2 * L + alt) * slat;
69 
70  return pi::Point3d(x, y, z);
71  }

73  static pi::Point3d XYZ2GPS(double x,double y,double z)
74  {
75  const double b = sqrt(WGS84_A*WGS84_A*(1-WGS84_E*WGS84_E));
76  const double ep = sqrt((WGS84_A*WGS84_A-b*b)/(b*b));
77  const double p = sqrt(x*x+y*y);
78  const double th = atan2(WGS84_A*z,b*p);
79  const double lon = atan2(y,x);
80  const double lat = atan2((z+ep*ep*b* pow(sin(th),3)),(p-WGS84_E*WGS84_E*WGS84_A*pow(cos(th),3)));
81  const double N = WGS84_A/sqrt(1-WGS84_E*WGS84_E*sin(lat)*sin(lat));
82  const double alt = p/cos(lat)-N;
83 
84  return pi::Point3d(R2D(lat), R2D(lon), alt);
85  }
